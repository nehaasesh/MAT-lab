package com.example.randomnumbergenerator

import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.widget.*
import kotlin.random.Random

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val minValue = findViewById<EditText>(R.id.minValue)
        val maxValue = findViewById<EditText>(R.id.maxValue)
        val generateBtn = findViewById<Button>(R.id.generateBtn)
        val resultText = findViewById<TextView>(R.id.resultText)

        generateBtn.setOnClickListener {

            val min = minValue.text.toString().toIntOrNull()
            val max = maxValue.text.toString().toIntOrNull()

            if (min == null || max == null) {
                resultText.text = "Please enter valid numbers"
                return@setOnClickListener
            }

            if (min > max) {
                resultText.text = "Minimum should be less than Maximum"
                return@setOnClickListener
            }

            val randomNumber = Random.nextInt(min, max + 1)
            resultText.text = "Random Number: $randomNumber"
        }
    }
}
